    Instrumental Broadening Quickstart¶

    This quickstart compares two instrumental broadening models on the same spectrum:

    1. Constant resolving power (R = constant)
    2. JWST NIRSpec broadening (R=R(wavelength))

    Gaussian kernel for instrumental broadening¶

    The normalized Gaussian kernel is

    $$ K(x) = \exp\left(-\frac{x^2}{2\sigma_{\rm pix}^2}\right), \qquad K \leftarrow \frac{K}{\sum K}, $$

    where $\sigma_{\rm pix}$ is the kernel width in pixels.

    For a resolving-power profile (R($\lambda$)), broadpy converts to velocity FWHM using

    $$ v_{\rm FWHM}(\lambda) = \frac{c}{R(\lambda)}, $$

    then to pixel sigma (on the internal log-wavelength sampling) with

    $$ \sigma_{\rm pix}(\lambda) = \frac{v_{\rm FWHM}(\lambda)}{c\,\sqrt{8\ln 2}\,\Delta\ln\lambda}. $$

    For variable broadening, you pass a vector fwhm (km/s), and InstrumentalBroadening(..., kernel="gaussian_variable") applies a local Gaussian kernel at each wavelength sample.

    # Load a JWST-like example spectrum in the NIRSpec G140H range.
wavelength_nm, flux = load_example_data(wave_range=(1000.0, 1600.0), jwst=True)
flux /= np.nanmedian(flux)
# 1) Constant-resolution broadening.
resolution_constant = 2700.0
instrument = InstrumentalBroadening(wavelength_nm, flux)
flux_constant_resolution = instrument(resolution=resolution_constant, kernel="gaussian")

# 2) JWST NIRSpec broadening.
flux_nirspec = nirspec_broadening(wavelength_nm, flux, grating="g140h")

residual = flux_nirspec - flux_constant_resolution

fig, (ax_resolution, ax_spectrum, ax_residuals) = plt.subplots(
    3,
    1,
    figsize=(14, 9),
    sharex=True,
    gridspec_kw={"height_ratios": [1, 3, 1]},
)

ax_resolution.plot(
    wavelength_nm,
    np.full_like(wavelength_nm, resolution_constant),
    label=f"Constant R={resolution_constant:.0f}",
)

# Effective R(lambda) used by the NIRSpec polynomial model.
resolution_nirspec = np.polyval(nirspec_model.poly_coefficients["g140h"], wavelength_nm)
ax_resolution.plot(wavelength_nm, resolution_nirspec, label="NIRSpec R(lambda)")
ax_resolution.set_title("(a) Resolution models")
ax_resolution.set_ylabel(r"R = $\lambda/\Delta\lambda$")
ax_resolution.legend(loc="best")

ax_spectrum.plot(
    wavelength_nm,
    flux_constant_resolution,
    linewidth=1.2,
    label=f"Constant R={resolution_constant:.0f}",
)

ax_spectrum.plot(
    wavelength_nm,
    flux_nirspec,
    linewidth=1.2,
    label="JWST NIRSpec (G140H)",
)
ax_spectrum.set_title("(b) Broadened spectra")
ax_spectrum.set_ylabel("Flux")
ax_spectrum.legend(loc="best")

ax_residuals.plot(wavelength_nm, residual, linewidth=1.0, label=r"NIRSpec R($\lambda$) - constant R")

ax_residuals.set_title("(c) Residuals")

ax_residuals.set_ylabel("Residual flux")
ax_residuals.legend(loc="best")

    # Load and plot JWST NIRSpec resolution profiles.

fig, ax = plt.subplots(1, 1, figsize=(12, 4))

for grating in NIRSPec.available_gratings:
    wavelength_range = NIRSPec.wavelength_range_gratings[grating]
    wavelength_grating = np.linspace(wavelength_range[0], wavelength_range[1], 1000)
    resolution = NIRSPec(wavelength=wavelength_grating).get_resolution_curve(grating=grating)
    ax.plot(wavelength_grating, resolution, linewidth=1.5, label=grating)

ax.set_xlabel("Wavelength [nm]")
ax.set_ylabel(r"Resolution $R = \lambda/\Delta\lambda$")
ax.set_title("JWST NIRSpec resolution profiles")
ax.legend(loc="best")
plt.tight_layout()
plt.show()
